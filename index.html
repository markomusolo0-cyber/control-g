This dashboard can receive sensor data directly from your ESP8266 over Wi-Fi. Configure your ESP8266 to POST JSON to the endpoint shown below.
ESP8266 Arduino sketch (Wi-Fi POST)
#include <ESP8266WiFi.h>
#include <ESP8266HTTPClient.h>
#include <ArduinoJson.h>

const char* ssid     = "YOUR_WIFI_SSID";
const char* password = "YOUR_WIFI_PASSWORD";
const char* endpoint = "http://<your-server-ip>:5000/data";

float readLevel()  { /* HC-SR04 ultrasonic */ return level_cm; }
float readFlow()   { /* YF-S201 pulse count */ return flow_Lmin; }
float readTemp()   { /* DS18B20 one-wire    */ return temp_C; }

void setup() {
  Serial.begin(115200);
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(500);
  Serial.println("Connected: " + WiFi.localIP().toString());
}

void loop() {
  StaticJsonDocument<256> doc;
  doc["level"] = readLevel();
  doc["fin"]   = readFlow();
  doc["fout"]  = 0;
  doc["temp"]  = readTemp();
  doc["pwm"]   = analogRead(A0) / 10.24;
  doc["sp"]    = 20.0;
  doc["mode"]  = "PID";

  String body; serializeJson(doc, body);
  WiFiClient client; HTTPClient http;
  http.begin(client, endpoint);
  http.addHeader("Content-Type", "application/json");
  int code = http.POST(body);
  Serial.println("POST " + String(code));
  http.end();
  delay(1000);
}
Python Flask receiver (run on your laptop)
from flask import Flask, request, jsonify
import csv, time

app = Flask(__name__)
log_file = "sensor_log.csv"

@app.route("/data", methods=["POST"])
def receive():
    data = request.get_json()
    data["ts"] = time.strftime("%H:%M:%S")
    print(data)
    with open(log_file, "a", newline="") as f:
        w = csv.DictWriter(f, fieldnames=data.keys())
        if f.tell() == 0: w.writeheader()
        w.writerow(data)
    return jsonify({"ok": True})

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=5000)

# Run: python server.py
# ESP8266 posts to: http://<laptop-ip>:5000/data
